module ram(data_in,read_en,write_en,clk,reset,data_out,addr);
  input [31:0] data_in;
  input clk,reset;
  input read_en,write_en;
  output reg[31:0] data_out;
  input [5:0] addr;
  
  reg [31:0] ram[31:0];
  integer i ;
  always @ (posedge clk)
    begin
      if(reset)
        begin
          if (reset) 
            begin
              ram <= '{default:32'd0};
          	  data_out <=0;
            end
        end
      else begin
        if(read_en)
          begin
            data_out <= ram[addr];
          end
        else if(write_en)
          begin
            ram[addr] <= data_in;
          end
        else 
          begin
            data_out <= data_out;
          end
        
      end
    end
endmodule
