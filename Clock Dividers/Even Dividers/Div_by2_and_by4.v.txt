module test(clk,reset,out);
  input clk;
  output out;
  input reset;
  reg [2:0] temp;
  
  wire out2; // to monitor div by 4 output.
  
  always @(posedge clk)
    begin
      if(reset) 
        temp <= 0;
       else 
         begin
          if (temp == 8)
            temp <= 0;
          else
            temp <= temp +1;     
         end 
      end
  assign out = temp[0];
  assign out2 = temp[1];
endmodule
