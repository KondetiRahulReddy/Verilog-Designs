module fib(clk,reset,out);
  input clk;
  input reset;
  output [10:0] out;
  
  reg [10:0] previous_value;
  reg [10:0] current_value;
  
  always@(posedge clk)
    if (reset) begin
      previous_value <= 0;
      current_value  <= 1;
    end
  else begin
    previous_value <= current_value;
    current_value <= previous_value + current_value;
  end
  assign out = previous_value;
endmodule
