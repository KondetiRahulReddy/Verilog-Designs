module test;
  reg clk,reset;
  wire [10:0] out;
  fib dut(.clk(clk),
          .reset(reset),
          .out(out));
  
  initial begin
    clk <= 0;
    reset <=1;
  end
  
  always #5 clk = ~clk;
  
  initial begin
    #5 reset <= 0; 
  end
  
  initial
    #100 $finish();
  
  
  initial begin
     $monitor("time = %d,fib_output = %d ", $time, out);
	$dumpfile("dump.vcd"); $dumpvars;
  end
endmodule
