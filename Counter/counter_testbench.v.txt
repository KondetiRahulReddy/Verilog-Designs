// Tb for 4-bit up counter.
module up_counter_tb();
  reg clk,reset;
  wire[3:0] counter_out;
  
  up_counter counter_inst(clk,reset,counter_out);
  
  initial begin
    clk = 0;
    forever #5 clk = ~clk;
  end
  
    initial begin
      reset = 1;
      #20 reset = 0;
    end
  
  initial begin
    $dumpfile("dump.vcd");
    $dumpvars;
  end
 initial  
   #200 $finish;
endmodule
